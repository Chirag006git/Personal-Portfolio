  ]
    \begin{tabularx}{\linewidth}{@{}X r@{}}
      {\bfseries\color{Dark}\small #1} &
      {\itshape\color{Muted}\scriptsize #3}\\[-2pt]
      {\color{Teal}\scriptsize\itshape #2} &

    \begin{itemize}[leftmargin=1.1em,itemsep=0pt,parsep=0pt,topsep=2pt]
      \small\color{Dark} #4
    \end{itemize}
